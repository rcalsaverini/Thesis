% fix bug with hyperref that doesn't link to the figure but the section it belongs to.
\AtBeginEnvironment{figure}{\phantomsection}
\AtBeginEnvironment{table}{\phantomsection}

%Show number in chapters, sections and subsections
\setcounter{secnumdepth}{2}

%Show sections and subsections on TOC
\setcounter{tocdepth}{2}

% Adjust number position in chapters and a a \rceil symbol to separate the number from the title
\titleformat{\chapter}%
  [hang]% shape
  {\relax}% format applied to label+text
  {\itshape\scshape\huge$\lceil$\thechapter$\rceil$}% label
  {15pt}% horizontal separation between label and title body
  {\huge\rmfamily\itshape\scshape}% before the title body
  []% after the title body

\titleformat{\section}%
  [hang]% shape
  {\relax}% format applied to label+text
  {\itshape\scshape\Large$\lceil$\thesection$\rceil$}% label
  {10pt}% horizontal separation between label and title body
  {\Large\rmfamily\itshape\scshape}% before the title body
  []% after the title body

\titleformat{\subsection}%
  [hang]% shape
  {\relax}% format applied to label+text
  {\itshape\scshape\large$\lceil$\thesubsection$\rceil$}% label
  {8pt}% horizontal separation between label and title body
  {\large\rmfamily\itshape\scshape}% before the title body
  []% after the title body
